  86. function _mkService($uris = null, $type_uris = null,
  87.                     $delegate = null, $dent = '        ')
  88. {
  89.     $chunks = array($dent, "<Service>\n");
  90.     $dent2 = $dent . '    ';
  91.     if ($type_uris) {
  92.         foreach ($type_uris as $type_uri) {
  93.             $chunks = array_merge($chunks,
  94.                                   array($dent2 . '<Type>',
  95.                                         $type_uri, "</Type>\n"));
  96.         }
  97.     }
  98.  
  99.     if ($uris) {
  100.         foreach ($uris as $uri) {
  101.             if (is_array($uri)) {
  102.                 list($uri, $prio) = $uri;
  103.             } else {
  104.                 $prio = null;
  105.             }
  106.             $chunks = array_merge($chunks, array($dent2, '<URI'));
  107.             if ($prio !== null) {
  108.                     $chunks = array_merge($chunks, array(' priority="', strval($prio), '"'));
  109.             }
  110.             $chunks = array_merge($chunks, array('>', $uri, "</URI>\n"));
  111.         }
  112.     }
  113.  
  114.     if ($delegate) {
  115.         $chunks = array_merge($chunks,
  116.                               array($dent2, '<openid:Delegate>',
  117.                                     $delegate, "</openid:Delegate>\n"));
  118.     }
  119.  
  120.     $chunks = array_merge($chunks, array($dent, "</Service>\n"));
  121.  
  122.     return implode("", $chunks);
  123. }
