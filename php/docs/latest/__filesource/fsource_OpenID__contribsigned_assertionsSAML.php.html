  45.    {

  51.       $samlResponse = str_replace('NOT_ON_OR_AFTER', $this->samlGetDateTime(strtotime($notOnOrAfter)),$samlResponse);
  52.       $samlResponse = str_replace('ASSERTION_ID',$this->samlCreateId(), $samlResponse);
  53.       $samlResponse = str_replace('RSADSA', strtolower($rsadsa), $samlResponse);
  54.       $samlResponse = str_replace('ISSUER_DOMAIN', $acsURI, $samlResponse);
  55.       $samlResponse = str_replace('ATTRIBUTE_NAME', $attribute, $samlResponse);

  203.    {

  205.       $indexT = strpos($samlDate, 'T');

  207.       if (($indexT != 10) || ($indexZ != 19)) {

  209.       }
  210.       $dateString = substr($samlDate, 0, 10);
  211.       $timeString = substr($samlDate, $indexT + 1, 8);
  212.       list($year, $month, $day) = explode('-', $dateString);
  213.       list($hour, $minute, $second) = explode(':', $timeString);
  214.       $parsedDate = gmmktime($hour, $minute, $second, $month, $day, $year);
  215.       if (($parsedDate === false) || ($parsedDate == -1)) return false;
  216.       if (!checkdate($month, $day, $year)) return false;
  217.       return $parsedDate;
  218.    }
