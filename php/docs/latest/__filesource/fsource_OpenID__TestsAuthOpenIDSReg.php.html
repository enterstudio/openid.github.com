  424.     function test_requestField()
  425.     {
  426.         global $Auth_OpenID_sreg_data_fields;
  427.  
  428.         // Add all of the fields, one at a time
  429.         $req = Auth_OpenID_SRegRequest::build();
  430.         $fields = array_keys($Auth_OpenID_sreg_data_fields);
  431.         foreach ($fields as $field_name) {
  432.             $req->requestField($field_name);
  433.         }
  434.  
  435.         $this->assertEquals($fields, $req->optional);
  436.         $this->assertEquals(array(), $req->required);
  437.  
  438.         // By default, adding the same fields over again has no effect
  439.         foreach ($fields as $field_name) {
  440.             $req->requestField($field_name);
  441.         }
  442.  
  443.         $this->assertEquals($fields, $req->optional);
  444.         $this->assertEquals(array(), $req->required);
  445.  
  446.         // Requesting a field as required overrides requesting it as
  447.         // optional
  448.         $expected = $fields;
  449.         $overridden = array_pop($expected);
  450.  
  451.         $this->assertTrue($req->requestField($overridden, true));
  452.  
  453.         $this->assertEquals($expected, $req->optional);
  454.         $this->assertEquals(array($overridden), $req->required);
  455.  
  456.         // Requesting a field as required overrides requesting it as
  457.         // optional
  458.         foreach ($fields as $field_name) {
  459.             $this->assertTrue($req->requestField($field_name, true));
  460.         }
  461.  
  462.         $this->assertEquals(array(), $req->optional);
  463.         foreach ($fields as $f) {
  464.             $this->assertTrue(in_array($f, $req->required));
  465.         }
  466.  
  467.         // Requesting it as optional does not downgrade it to optional
  468.         foreach ($fields as $field_name) {
  469.             $req->requestField($field_name);
  470.         }
  471.  
  472.         $this->assertEquals(array(), $req->optional);
  473.  
  474.         foreach ($fields as $f) {
  475.             $this->assertTrue(in_array($f, $req->required));
  476.         }

  486.     {

  488.

  490.         $req = Auth_OpenID_SRegRequest::build();

  494.  
  495.         $this->assertEquals($fields, $req->optional);
  496.         $this->assertEquals(array(), $req->required);
  497.  
  498.         // By default, adding the same fields over again has no effect
  499.         $req->requestFields($fields);

  501.         $this->assertEquals($fields, $req->optional);
  502.         $this->assertEquals(array(), $req->required);
  503.  
  504.         // Requesting a field as required overrides requesting it as
  505.         // optional

  507.         $overridden = array_shift($expected);
  508.         $req->requestFields(array($overridden), true);

  510.         foreach ($expected as $f) {
  511.             $this->assertTrue(in_array($f, $req->optional));
  512.         }
  513.  
  514.         $this->assertEquals(array($overridden), $req->required);
  515.  
  516.         // Requesting a field as required overrides requesting it as
  517.         // optional
  518.         $req->requestFields($fields, true);
  519.  
  520.         $this->assertEquals(array(), $req->optional);
  521.         $this->assertEquals($fields, $req->required);
  522.  
  523.         // Requesting it as optional does not downgrade it to optional
  524.         $req->requestFields($fields);
  525.  
  526.         $this->assertEquals(array(), $req->optional);
