  345.     function _testNonceCleanup($store) {
  346.         if (!$store->supportsCleanup()) {
  347.             return;
  348.         }
  349.  
  350.         $server_url = 'http://www.myopenid.com/openid';
  351.  
  352.         $now = time();

  354.         $old_nonce1 = Auth_OpenID_mkNonce($now - 20000);
  355.         $old_nonce2 = Auth_OpenID_mkNonce($now - 10000);
  356.         $recent_nonce = Auth_OpenID_mkNonce($now - 600);
  357.  
  358.         global $Auth_OpenID_SKEW;
  359.         $orig_skew = $Auth_OpenID_SKEW;
  360.  
  361.         $Auth_OpenID_SKEW = 0;
  362.         $store->cleanupNonces();
  363.         // Set SKEW high so stores will keep our nonces.
  364.         $Auth_OpenID_SKEW = 100000;
  365.  
  366.         $params = Auth_OpenID_splitNonce($old_nonce1);
  367.         array_unshift($params, $server_url);
  368.         $this->assertTrue(call_user_func_array(array($store, 'useNonce'), $params));
  369.  
  370.         $params = Auth_OpenID_splitNonce($old_nonce2);
  371.         array_unshift($params, $server_url);
  372.         $this->assertTrue(call_user_func_array(array($store, 'useNonce'), $params));
  373.  
  374.         $params = Auth_OpenID_splitNonce($recent_nonce);
  375.         array_unshift($params, $server_url);
  376.         $this->assertTrue(call_user_func_array(array($store, 'useNonce'), $params));
  377.  
  378.         $Auth_OpenID_SKEW = 3600;
  379.         $cleaned = $store->cleanupNonces();
  380.         $this->assertEquals(2, $cleaned); // , "Cleaned %r nonces." % (cleaned,)
  381.  
  382.         $Auth_OpenID_SKEW = 100000;
  383.         // A roundabout method of checking that the old nonces were
  384.         // cleaned is to see if we're allowed to add them again.
  385.  
  386.         $params = Auth_OpenID_splitNonce($old_nonce1);
  387.         array_unshift($params, $server_url);
  388.         $this->assertTrue(call_user_func_array(array($store, 'useNonce'), $params));
  389.         $params = Auth_OpenID_splitNonce($old_nonce2);
  390.         array_unshift($params, $server_url);
  391.         $this->assertTrue(call_user_func_array(array($store, 'useNonce'), $params));
  392.  
  393.         // The recent nonce wasn't cleaned, so it should still fail.

  395.         array_unshift($params, $server_url);

  397.  
  398.         $Auth_OpenID_SKEW = $orig_skew;
