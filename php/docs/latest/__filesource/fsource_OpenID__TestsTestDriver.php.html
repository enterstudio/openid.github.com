  49. /**
  50.  * Load the tests that are defined in the named modules.
  51.  *
  52.  * If you have Tests/Foo.php which defines a test class called
  53.  * Tests_Foo, the call would look like:
  54.  *
  55.  * loadTests('Tests/', array('Foo'))
  56.  *
  57.  * @param string $test_dir The root of the test hierarchy. Must end
  58.  *  with a /
  59.  *
  60.  * @param array $test_names The names of the modules in which the
  61.  *  tests are defined. This should not include the root of the test
  62.  *  hierarchy.
  63.  */
  64. function loadTests($test_dir, $test_names)
  65. {
  66.     global $_tests;
  67.     $suites = array();
  68.  
  69.     foreach ($test_names as $filename) {
  70.         $filename = $test_dir . $filename . '.php';
  71.  
  72.         if (!global_require_once($filename)) {
  73.             continue;
  74.         }
  75.  
  76.         $class_name = str_replace('/', '_', $filename);
  77.         $class_name = basename($class_name, '.php');
  78.  
  79.         $suites[] = makeSuite($class_name);
  80.     }
  81.  
  82.     return $suites;
  83. }

  85. function makeSuite($class_name) {
  86.     $test = new $class_name($class_name);
  87.  
  88.     if (is_a($test, 'PHPUnit_Framework_TestCase')) {
  89.         $s = new PHPUnit_Framework_TestSuite();
  90.         $s->setName($class_name);
  91.         $s->addTestSuite($class_name);
  92.         $test = $s;
  93.     }
  94.  
  95.     $tc_array_name = $class_name . '_other';
  96.     if (array_key_exists($tc_array_name, $GLOBALS) &&
  97.         is_array($GLOBALS[$tc_array_name])) {
  98.  
  99.         foreach ($GLOBALS[$tc_array_name] as $tc) {
  100.             $test->addTestSuite(get_class($tc));
  101.         }
  102.     }
  103.  
  104.     return $test;
  105. }

  108. function global_require_once($name)
  109. {
  110.     $f = include_once $name;
  111.     if (!$f) {
  112.         print("global require once skipping $name\n");
  113.         return false;
  114.     }
  115.     foreach (get_defined_vars() as $k => $v) {
  116.         if (!in_array($k, array('name', 'GLOBALS'))) {
  117.             $GLOBALS[$k] = $v;
  118.         }
  119.     }
  120.     return true;

  166. function selectTests($package, $names)
  167. {
  168.     global $_tests;
  169.     $lnames = array_map('strtolower', $names);
  170.     $include = array();
  171.     $exclude = array();
  172.     foreach ($package['files'] as $t) {
  173.         $l = strtolower($t);
  174.         if (in_array($l, $lnames)) {
  175.             $include[] = $t;
  176.         }
  177.  
  178.         if (in_array("/$l", $lnames)) {
  179.             $exclude[] = $t;
  180.         }
  181.     }
  182.  
  183.     return array_diff($include, $exclude);
  184. }

  187. function loadSuite($names=null)
  188. {
  189.     global $_tests;
  190.     $result = array();
  191.     foreach ($_tests as $package) {
  192.         if (!$names) {
  193.             $selected = $package['files'];
  194.         } else {
  195.             $selected = selectTests($package, $names);
  196.         }
  197.         $result = array_merge($result, loadTests($package['dir'], $selected));
  198.     }
  199.  
  200.     return $result;
  201. }
