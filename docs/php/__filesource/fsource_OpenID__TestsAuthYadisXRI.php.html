  97. class Tests_Auth_Yadis_TestGetRootAuthority extends PHPUnit_Framework_TestCase {
  98.     function runTest()
  99.     {
  100.         $xris = array(
  101.                       array("@foo", "@"),

  112.                       // Looking at the ABNF in XRI Syntax 2.0, I don't think you can
  113.                       // have example.com*bar.  You can do (example.com)*bar, but that
  114.                       // would mean something else.
  115.                       // ("example.com*bar/(=baz)", "example.com*bar"),
  116.                       // ("baz.example.com!01/foo", "baz.example.com!01"),
  117.                       );
  118.  
  119.         foreach ($xris as $tupl) {
  120.             list($thexri, $expected_root) = $tupl;
  121.             $this->assertEquals(Auth_Yadis_XRI($expected_root),
  122.                                 Auth_Yadis_rootAuthority($thexri),
