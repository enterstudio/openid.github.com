  9. /**
  10.  * A consumer whose _requestAssocation will return predefined results
  11.  * instead of trying to actually perform association requests.
  12.  */
  13. class ErrorRaisingConsumer extends Auth_OpenID_GenericConsumer {
  14.     // The list of objects to be returned by successive calls to
  15.     // _requestAssocation.  Each call will pop the first element from
  16.     // this list and return it to _negotiateAssociation.  If the
  17.     // element is a Message object, it will be wrapped in a
  18.     // ServerErrorContainer exception.  Otherwise it will be returned
  19.     // as-is.
  20.     var $return_messages = array();
  21.  
  22.     function _requestAssociation($endpoint, $assoc_type, $session_type)
  23.     {
  24.         $m = array_pop($this->return_messages);
  25.         if (is_a($m, 'Auth_OpenID_Message')) {
  26.             return Auth_OpenID_ServerErrorContainer::fromMessage($m);
  27.         } else if (Auth_OpenID::isFailure($m)) {
  28.             return $m;
  29.         } else {
  30.             return $m;
  31.         }
  32.     }
  33. }
