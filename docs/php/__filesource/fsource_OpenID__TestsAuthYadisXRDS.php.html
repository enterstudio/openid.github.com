  179.         function _URIMatchesSchtuff($service)

  191.         // This filter should only match one.
  192.         function _URIMatchesMyOpenID($service)
  193.             {
  194.                 $uris = $service->getURIs();
  195.  
  196.                 foreach ($uris as $uri) {
  197.                     if (preg_match("|myopenid|", $uri)) {
  198.                         return true;

  200.                 }
  201.                 return false;
  202.             }

  204.         // Make sure a pair of filters in ALL mode only match one service.

  207.                                                     "_URIMatchesSchtuff"),

  209.                             "_HasTypeAndURI / _URIMatchesSchtuff check");

  211.         // Make sure a pair of filters in ALL mode only match one service.
  212.         $this->assertEquals(count(
  213.                               $xrds->services(array("_HasTypeAndURI",
  214.                                                     "_URIMatchesMyOpenID"),
