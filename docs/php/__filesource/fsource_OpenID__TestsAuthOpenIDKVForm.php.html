  58.     function Tests_Auth_OpenID_KVForm_TestCase_Parse(
  59.         $arr, $str, $lossy, $errs)
  60.     {
  61.  
  62.         $this->arr = $arr;
  63.         $this->str = $str;
  64.         $this->lossy = $lossy;
  65.         $this->errs = $errs;
  66.     }
  67.  
  68.     function _runTest()
  69.     {
  70.         // Do one parse, after which arrayToKV and kvToArray should be
  71.         // inverses.
  72.         $parsed1 = Auth_OpenID_KVForm::toArray($this->str);
  73.         $serial1 = Auth_OpenID_KVForm::fromArray($this->arr);
  74.  
  75.         if ($this->lossy == "neither" || $this->lossy == "str") {
  76.             $this->assertEquals($this->arr, $parsed1, "str was lossy");
  77.         }
  78.  
  79.         if ($this->lossy == "neither" || $this->lossy == "arr") {
  80.             $this->assertEquals($this->str, $serial1, "array was lossy");
  81.         }
  82.  
  83.         $parsed2 = Auth_OpenID_KVForm::toArray($serial1);
  84.         $serial2 = Auth_OpenID_KVForm::fromArray($parsed1);
  85.  
  86.         // Round-trip both
  87.         $parsed3 = Auth_OpenID_KVForm::toArray($serial2);
  88.         $serial3 = Auth_OpenID_KVForm::fromArray($parsed2);
  89.  
  90.         $this->assertEquals($serial2, $serial3, "serialized forms differ");
  91.  
  92.         // Check to make sure that they're inverses.
  93.         $this->assertEquals($parsed2, $parsed3, "parsed forms differ");
  94.  
  95.     }
